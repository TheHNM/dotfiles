    149 parse_git_branch() {
    150   # Show Git branch/tag, or name-rev if on detached head
    151   ( git symbolic-ref -q HEAD || git name-rev --name-only --no-undefined --always HEAD ) 2> /dev/null
    152 }

    154 parse_git_state() {
    155   # Show different symbols as appropriate for various Git repository states
    156   # Compose this value via multiple conditional appends.
    157   local GIT_STATE=""
    158   local NUM_AHEAD="$(git log --oneline @{u}.. 2> /dev/null | wc -l | tr -d ' ')"
    159   if [ "$NUM_AHEAD" -gt 0 ]; then
    160     GIT_STATE=$GIT_STATE${GIT_PROMPT_AHEAD//NUM/$NUM_AHEAD}
    161   fi
    162   local NUM_BEHIND="$(git log --oneline ..@{u} 2> /dev/null | wc -l | tr -d ' ')"
    163   if [ "$NUM_BEHIND" -gt 0 ]; then
    164     GIT_STATE=$GIT_STATE${GIT_PROMPT_BEHIND//NUM/$NUM_BEHIND}
    165   fi
    166   local GIT_DIR="$(git rev-parse --git-dir 2> /dev/null)"
    167   if [ -n $GIT_DIR ] && test -r $GIT_DIR/MERGE_HEAD; then
    168     GIT_STATE=$GIT_STATE$GIT_PROMPT_MERGING
    169   fi
    170   if [[ -n $(git ls-files --other --exclude-standard 2> /dev/null) ]]; then
    171     GIT_STATE=$GIT_STATE$GIT_PROMPT_UNTRACKED
    172   fi
    173   if ! git diff --quiet 2> /dev/null; then
    174     GIT_STATE=$GIT_STATE$GIT_PROMPT_MODIFIED
    175   fi
    176   if ! git diff --cached --quiet 2> /dev/null; then
    177     GIT_STATE=$GIT_STATE$GIT_PROMPT_STAGED
    178   fi
    179   if [[ -n $GIT_STATE ]]; then
    180     echo "$GIT_PROMPT_PREFIX$GIT_STATE$GIT_PROMPT_SUFFIX"
    181   fi
    182 }

    184 git_prompt_string() {
    185   local git_where="$(parse_git_branch)"

    187   # If inside a Git repository, print its branch and state
    188   [ -n "$git_where" ] && echo "$GIT_PROMPT_SYMBOL$(parse_git_state)$GIT_PROMPT_PREFIX%{$fg[yellow]%}${git_where#(refs/heads/|tags/)}$GIT_PROMPT_SUFFIX"
    189 
    190   # If not inside the Git repo, print exit codes of last command (only if it failed)
    191   [ ! -n "$git_where" ] && echo "%{$fg[red]%} %(?..[%?])"
    192 }
