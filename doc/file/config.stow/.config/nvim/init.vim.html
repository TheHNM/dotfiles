    109 " Cycle through completions with TAB (and SHIFT-TAB cycles backwards)
    110 function! InsertTabWrapper(direction)
    111     let col = col('.') - 1
    112     if !col || getline('.')[col - 1] !~ '\k'
    113         return "\<tab>"
    114     elseif "backward" == a:direction
    115         return "\<c-p>"
    116     else
    117         return "\<c-n>"
    118     endif
    119 endfunction

    121 inoremap <s-tab> <c-r>=InsertTabWrapper ("backward")<cr>
